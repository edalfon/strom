import statsmodels.api as sm from sklearn.base import BaseEstimator, RegressorMixin

class SMWrapper(BaseEstimator, RegressorMixin): âââA universal sklearn-style wrapper for statsmodels regressorsâââ

def __init__(self, model_class, fit_intercept=True):
    self.model_class = model_class
    self.fit_intercept = fit_intercept

def fit(self, X, y):
    if self.fit_intercept:
        X = sm.add_constant(X)
    self.model_ = self.model_class(y, X)
    self.results_ = self.model_.fit()
    return self

def predict(self, X):
    if self.fit_intercept:
        X = sm.add_constant(X)
    return self.results_.predict(X)
